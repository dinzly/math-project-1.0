    <script>
        // Generate dataset based on logistic growth formula
        const labels = [];
        const yieldData = [];
        const derivativeData = [];

        for (let x = 1; x <= 20; x += 0.5) {
            labels.push(x);
            
            // f(x) Logistic curve formula
            const fx = 30 / (1 + Math.exp(-0.4 * (x - 7)));
            yieldData.push(fx.toFixed(2));

            // f'(x) Derivative formula
            const expTerm = Math.exp(-0.4 * (x - 7));
            const fPrimeX = (12 * expTerm) / Math.pow(1 + expTerm, 2);
            derivativeData.push(fPrimeX.toFixed(2));
        }

        // Initialize Chart.js
        const ctx = document.getElementById('yieldChart').getContext('2d');
        const yieldChart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: labels,
                datasets: [{
                    label: 'Cumulative FFB Yield f(x) (tonnes/hectare)',
                    data: yieldData,
                    borderColor: '#2563eb',
                    backgroundColor: 'rgba(37, 99, 235, 0.1)',
                    borderWidth: 3,
                    fill: true,
                    tension: 0.3,
                    pointRadius: 3,
                    pointHoverRadius: 8,
                    pointHoverBackgroundColor: '#0d9488'
                }]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                    legend: { position: 'top' },
                    tooltip: {
                        callbacks: {
                            label: function(context) {
                                return `Yield: ${context.parsed.y} tonnes/ha`;
                            }
                        }
                    }
                },
                scales: {
                    x: {
                        title: { display: true, text: 'Tree Age (Years Since Planting)', font: { weight: 'bold' } }
                    },
                    y: {
                        title: { display: true, text: 'Yield (tonnes per Hectare)', font: { weight: 'bold' } },
                        suggestedMax: 35
                    }
                },
                // Trigger action when clicking a data point on the graph
                onClick: (e, elements) => {
                    if (elements.length > 0) {
                        const index = elements[0].index;
                        const age = labels[index];
                        const currentYield = yieldData[index];
                        const currentRate = derivativeData[index];

                        displayCalculusInterpretation(age, currentYield, currentRate);
                    }
                }
            }
        });

        // Function to handle the interactive layman explanation updates
        function displayCalculusInterpretation(age, yieldVal, rateVal) {
            document.getElementById('instructionText').classList.add('hidden');
            const dataContent = document.getElementById('dataContent');
            dataContent.classList.remove('hidden');

            document.getElementById('outAge').innerText = `${age} Years`;
            document.getElementById('outYield').innerText = `${yieldVal} t/ha`;
            document.getElementById('outDerivative').innerText = `${rateVal} t/ha/yr`;

            let explanation = "";
            if (age < 5) {
                explanation = `The tree is young. The positive derivative of ${rateVal} tonnes/ha/year means the palm is in its exponential growth stage, building up canopy structure. Production accelerates every year.`;
            } else if (age >= 5 && age <= 9) {
                explanation = `At ${age} years old, the tree is approaching or at its peak biological metabolic efficiency. The derivative is at its highest point (${rateVal} t/ha/yr). This indicates the inflection point where the tree adds the maximum amount of fruit mass per year.`;
            } else if (age > 9 && age <= 15) {
                explanation = `The tree remains highly productive, but notice how the derivative is dropping to ${rateVal} tonnes/ha/year. Yield is still increasing, but it's increasing at a slower rate. The plant is reaching its physiological equilibrium.`;
            } else {
                explanation = `The plantation has reached absolute maturity. The derivative drops near zero (${rateVal} t/ha/yr), meaning the yield has flattened out. Economically, this tells agrotech managers that replanting schedules should be prepared soon as minimal new growth occurs.`;
            }

            document.getElementById('outExplanation').innerText = explanation;

            // Simple micro-animation effect for feedback
            const card = document.getElementById('outputCard');
            card.classList.remove('scale-100');
            card.classList.add('scale-102', 'border-2', 'border-emerald-500');
            setTimeout(() => {
                card.classList.remove('scale-102', 'border-2', 'border-emerald-500');
                card.classList.add('scale-100');
            }, 250);
        }
    </script>
